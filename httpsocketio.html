HTTP

Creating a node HTTP server

... in a few lines of code


var http = require('http');

var server = http.createServer();

server.on('request', function(req, res) {

  res.statusCode = 200;
  res.setHeader('Content-Type', 'text/plain');
  res.write('Hello World!');
  res.end();
});

server.listen(4000);

var inspect = require('util').inspect;

require('http').createServer(function(req, res) {

  res.statusCode = 200;
  res.setHeader('Content-Type', 'text/plain');

  var inspected = {
    'url': req.url,
    'method': req.method,
    'headers': req.headers
  };

  res.end(inspect(inspected));
}).listen(4001);

Gist #3

Piping requests

Request bodies are streams.


var fs = require('fs');

require('http').createServer(function(req, res) {
  res.writeHead(200, {'Content-Type': 'text/html'});

  if (req.method === 'GET') {
    fs.createReadStream(__dirname + '/form.html').pipe(res);
  } else {
    var fileName = __dirname + '/request_' + Date.now() + '.log';
    console.log('writing to ', fileName);
    var file = fs.createWriteStream(fileName);

    req.pipe(file);

    req.on('end', function() {
      res.end('Submitted, thanks');
    });
  }

}).listen(4001);

Gist #6

Streaming responses

... in chunks


var fs = require('fs');

require('http').createServer(function(req, res) {

  if (req.url === '/') {
    res.writeHead(200, {'Content-Type': 'video/x-m4v'});

    var rs = fs.createReadStream(__dirname + '/video.m4v');
    rs.pipe(res);

    rs.on('data', function(d) {
      console.log('sent', d.length, 'bytes')
    });
    rs.on('end', function() {
      console.log('read stream ended.');
    });

  } else {
    res.writeHead(404);
  }

}).listen(4000);

Shutting down a server

... after the first request


var server = require('http').createServer();
var closed = false;

server.on('request', function(req, res) {
  if (! closed) {
    res.end('Shutting down');
    server.close();
    closed = true;
  } else {
    res.end();
  }
});

server.listen(4000);

Socket.io events

Follows traditional event emitter pattern


var server = require('http').createServer(/*handler*/);
var io = require('socket.io').listen(server);
server.listen(8000);

io.sockets.on('connection', function (socket) {
  io.sockets.emit('this', { will: 'be received by everyone'});

  socket.on('private message', function (msg) {
    console.log('I received a private message:', msg);
  });

  socket.on('disconnect', function () {
    io.sockets.emit('user disconnected');
  });
});

Gist #13

Chat server

Time to play


var fs = require('fs');
var server = require('http').createServer(handler);
var io = require('socket.io').listen(server);

function handler (req, res) {
  if (req.url === '/') {
    res.statusCode = 200;
    res.setHeader('content-type', 'text/html');
    fs.createReadStream('./index.html').pipe(res);
  }
}

io.sockets.on('connection', function (socket) {
  socket.on('chat', function (data) {
    socket.get('username', function (err, name) {
      io.sockets.emit('echo', data, name);
    });
  });
  socket.on('setName', function (name) {
    socket.set('username', name, function () {
      socket.broadcast.emit('join', name);
    });
  });
});

server.listen(8080);
